<!-- drift.Hunk: OldStart, OldLines, NewStart, NewLines int; Lines []Line         -->
<!-- drift.Line: Op Op; Content string; OldNum, NewNum int                        -->
<!-- drift.Op: Equal=0, Insert=1, Delete=2                                        -->
<!--                                                                               -->
<!-- apply() algorithm:                                                             -->
<!--   Given DiffResult and original []string oldLines:                            -->
<!--   Walk all Line entries in all Hunks:                                         -->
<!--     - Equal lines: emit the line (it exists in both)                          -->
<!--     - Delete lines: skip (was removed)                                        -->
<!--     - Insert lines: emit the new line                                         -->
<!--   But also include lines NOT mentioned in any hunk (the Equal context not     -->
<!--   covered by hunks) by tracking which old line numbers appear as Equal.       -->
<!--   Result must equal the original newLines input.                              -->
<!--                                                                               -->
<!-- Simpler apply() approach:                                                     -->
<!--   Walk all edits (not hunks) from internal diff — but that requires access   -->
<!--   to internal state. Better: reconstruct from hunks by:                       -->
<!--   1. Sorting hunks by NewStart                                                 -->
<!--   2. For each hunk, emit Equal and Insert lines from hunk.Lines              -->

    ```
    For any old, new text:
    result, _ := drift.Diff(old, new)
    sum(h.OldLines for h in result.Hunks) <= len(oldLines)
    sum(h.NewLines for h in result.Hunks) <= len(newLines)
    ```

    ```go
    // Package testdata provides test helpers for drift property-based tests.
    package testdata

    import "github.com/tbcrawford/drift"

    // Apply reconstructs the "new" file from a DiffResult and the original "old" lines.
    // Used to verify the round-trip invariant: Apply(Diff(old, new), oldLines) == newLines.
    func Apply(result drift.DiffResult, oldLines []string) []string {
        if result.IsEqual {
            return oldLines
        }
        // Build a set of which old line numbers are covered by hunks
        // to know which old lines to pass through untouched.
        var out []string
        oldIdx := 0 // 0-indexed cursor through old lines

        for _, h := range result.Hunks {
            // Emit old lines before this hunk (equal, not in any hunk)
            hunkOldStart := h.OldStart - 1 // convert to 0-indexed
            for oldIdx < hunkOldStart {
                out = append(out, oldLines[oldIdx])
                oldIdx++
            }
            // Emit lines from the hunk
            for _, l := range h.Lines {
                switch l.Op {
                case drift.Equal:
                    out = append(out, l.Content)
                    oldIdx++
                case drift.Insert:
                    out = append(out, l.Content)
                case drift.Delete:
                    oldIdx++ // skip deleted line
                }
            }
        }
        // Emit remaining old lines after the last hunk
        for oldIdx < len(oldLines) {
            out = append(out, oldLines[oldIdx])
            oldIdx++
        }
        if out == nil {
            return []string{}
        }
        return out
    }
    ```
