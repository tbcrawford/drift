```
for each pair in pairs:
    oldContent := "" // empty string for added files
    newContent := ""
    oldName := "a/" + pair.Name
    newName := "b/" + pair.Name

    if pair.OldPath != "":
        read pair.OldPath → oldContent (return error on failure)
    if pair.NewPath != "":
        read pair.NewPath → newContent (return error on failure)

    result, err := drift.Diff(oldContent, newContent, opts.driftOpts...)
    if err: return false, wrapExitCode(err)
    if result.IsEqual: continue  // skip identical files

    hasDiff = true
    fmt.Fprintf(buf, "=== %s ===\n", pair.Name)
    if err := drift.RenderWithNames(result, buf, oldName, newName, opts.driftOpts...); err != nil:
        return false, newExitCode(2, err.Error())
```

**Extend `runRoot`** to detect directory args early, before calling `resolveInputs`:

```go
// Directory diff: both positional args are directories
if len(opts.args) == 2 && isDir(opts.args[0]) && isDir(opts.args[1]) {
    pairs, err := diffDirectories(opts.args[0], opts.args[1])
    if err != nil {
        return newExitCode(2, err.Error())
    }
    var buf bytes.Buffer
    hasDiff, err := runDirectoryDiff(pairs, opts, &buf)
    if err != nil {
        return err
    }
    if !hasDiff {
        return nil // identical dirs → exit 0
    }
    // Route through existing pager logic (same pattern as single-file path)
    var termHeight int
    if f, ok := opts.streams.Out.(*os.File); ok {
        if _, h, tErr := term.GetSize(f.Fd()); tErr == nil {
            termHeight = h
        }
    }
    lineCount := strings.Count(buf.String(), "\n")
    if shouldPage(opts.streams.Out, lineCount, termHeight, opts.noPager) {
        pagerWriter, cleanup, pErr := startPager(resolvePager(), opts.streams)
        if pErr != nil {
            _, _ = buf.WriteTo(opts.streams.Out)
        } else {
            _, _ = buf.WriteTo(pagerWriter)
            cleanup()
        }
    } else {
        _, _ = buf.WriteTo(opts.streams.Out)
    }
    return newExitCode(1, "")
}
```

Place this block at the TOP of `runRoot`, before the existing `resolveInputs` call.
