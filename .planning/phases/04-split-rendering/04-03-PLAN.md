## Objective

Add `WithSplit() Option` to the public API. Wire `drift.Render()` and `drift.RenderWithNames()` to resolve terminal width via `render.TerminalWidth(w)`, set `rcfg.TermWidth`, and route to `render.Split()` when `cfg.split == true`. Write integration tests that call `drift.Diff()` + `drift.Render(..., drift.WithSplit())` and verify two-panel output containing a separator.

## must_haves
- [ ] `options.go` has `split bool` field in `config` struct
- [ ] `options.go` exports `func WithSplit() Option` that sets `c.split = true`
- [ ] `render.go` resolves `termWidth := render.TerminalWidth(w)` before building `rcfg`
- [ ] `render.go` sets `rcfg.TermWidth = termWidth` in both `Render` and `RenderWithNames`
- [ ] `render.go` routes to `render.Split(result, wrapped, rcfg)` when `cfg.split == true`
- [ ] `drift.Render(..., drift.WithSplit())` produces output containing `│` separator
- [ ] `go test ./... -count=1` exits 0

```go
package drift_test

import (
	"bytes"
	"strings"
	"testing"
)

func TestRender_WithSplit_ContainsSeparator(t *testing.T) {
	old := "package main\n\nfunc hello() {\n\tfmt.Println(\"hello\")\n}\n"
	new_ := "package main\n\nfunc hello() {\n\tfmt.Println(\"hello, world\")\n\tfmt.Println(\"done\")\n}\n"

	result, err := Diff(old, new_)
	if err != nil {
		t.Fatalf("Diff error: %v", err)
	}

	var buf bytes.Buffer
	if err := Render(result, &buf, WithSplit(), WithNoColor()); err != nil {
		t.Fatalf("Render error: %v", err)
	}

	output := buf.String()
	if !strings.Contains(output, "|") {
		t.Errorf("WithSplit output must contain '|' separator; got:\n%s", output)
	}
}

func TestRender_WithSplit_ContainsBothPanelContents(t *testing.T) {
	old := "line one\nline two\n"
	new_ := "line one\nline THREE\n"

	result, err := Diff(old, new_)
	if err != nil {
		t.Fatalf("Diff error: %v", err)
	}

	var buf bytes.Buffer
	if err := Render(result, &buf, WithSplit(), WithNoColor()); err != nil {
		t.Fatalf("Render error: %v", err)
	}

	output := buf.String()
	if !strings.Contains(output, "line two") {
		t.Errorf("WithSplit output should contain old content 'line two'; got:\n%s", output)
	}
	if !strings.Contains(output, "line THREE") {
		t.Errorf("WithSplit output should contain new content 'line THREE'; got:\n%s", output)
	}
}

func TestRender_WithSplit_IdenticalInputs(t *testing.T) {
	old := "no changes here\n"
	result, err := Diff(old, old)
	if err != nil {
		t.Fatalf("Diff error: %v", err)
	}

	var buf bytes.Buffer
	if err := Render(result, &buf, WithSplit(), WithNoColor()); err != nil {
		t.Fatalf("Render(identical) error: %v", err)
	}
	// Identical inputs → no hunks → empty output
	if buf.Len() != 0 {
		t.Errorf("expected empty output for identical inputs, got: %q", buf.String())
	}
}

func TestRender_WithSplit_HunkHeaderFormat(t *testing.T) {
	old := "a\nb\n"
	new_ := "a\nc\n"

	result, err := Diff(old, new_)
	if err != nil {
		t.Fatalf("Diff error: %v", err)
	}

	var buf bytes.Buffer
	if err := Render(result, &buf, WithSplit(), WithNoColor()); err != nil {
		t.Fatalf("Render error: %v", err)
	}

	output := buf.String()
	if !strings.Contains(output, "@@ -") {
		t.Errorf("WithSplit output must contain '@@ -' hunk header; got:\n%s", output)
	}
}

func TestRender_WithoutSplit_StillProducesUnified(t *testing.T) {
	old := "x\n"
	new_ := "y\n"

	result, err := Diff(old, new_)
	if err != nil {
		t.Fatalf("Diff error: %v", err)
	}

	var buf bytes.Buffer
	if err := Render(result, &buf, WithNoColor()); err != nil {
		t.Fatalf("Render error: %v", err)
	}

	output := buf.String()
	// Without WithSplit, output should follow unified format (--- / +++ headers).
	if !strings.Contains(output, "---") {
		t.Errorf("non-split output should contain unified '---' header; got:\n%s", output)
	}
	if !strings.Contains(output, "+++") {
		t.Errorf("non-split output should contain unified '+++' header; got:\n%s", output)
	}
}
```
