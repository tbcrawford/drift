```go
// Package funcctx finds the enclosing function or class context for a hunk,
// matching git's -p funcname behavior. It scans backward from the hunk start
// to find the nearest line matching a language-specific pattern.
package funcctx

import (
    "regexp"
    "strings"
)

// maxLookback is the maximum number of lines to scan backward.
// Git uses 100 lines as the default; we match that.
const maxLookback = 100

// pattern holds a compiled regex for function/class context detection.
type pattern struct {
    re *regexp.Regexp
}

// langPatterns maps Chroma language names (lowercased) to a slice of patterns.
// Patterns are tested in order; the first match wins.
// These are inspired by git's userdiff.c built-in drivers.
var langPatterns = map[string][]*pattern{
    "go": {
        {re: regexp.MustCompile(`^func\s`)},
        {re: regexp.MustCompile(`^type\s`)},
    },
    "python": {
        {re: regexp.MustCompile(`^(async\s+)?def\s`)},
        {re: regexp.MustCompile(`^class\s`)},
    },
    "javascript": {
        {re: regexp.MustCompile(`^(export\s+)?(default\s+)?(async\s+)?function\s`)},
        {re: regexp.MustCompile(`^(export\s+)?(default\s+)?class\s`)},
        {re: regexp.MustCompile(`^\s*(async\s+)?[a-zA-Z_$][a-zA-Z0-9_$]*\s*[:=]\s*(async\s+)?(\([^)]*\)|[a-zA-Z_$][a-zA-Z0-9_$]*)\s*=>`)},
    },
    "typescript": {
        {re: regexp.MustCompile(`^(export\s+)?(default\s+)?(async\s+)?function\s`)},
        {re: regexp.MustCompile(`^(export\s+)?(abstract\s+)?class\s`)},
        {re: regexp.MustCompile(`^(export\s+)?(interface|type|enum)\s`)},
    },
    "c": {
        {re: regexp.MustCompile(`^[a-zA-Z_*][a-zA-Z0-9_*\s]+\(`)},
        {re: regexp.MustCompile(`^(struct|union|enum|typedef)\s`)},
    },
    "c++": {
        {re: regexp.MustCompile(`^[a-zA-Z_*][a-zA-Z0-9_*:<>\s]+\(`)},
        {re: regexp.MustCompile(`^(class|struct|namespace)\s`)},
    },
    "java": {
        {re: regexp.MustCompile(`^\s*(public|private|protected|static|final|abstract|synchronized|native|strictfp|\s)+[\w<>\[\]]+\s+\w+\s*\(`)},
        {re: regexp.MustCompile(`^\s*(public|private|protected)?\s*(abstract\s+)?(class|interface|enum)\s`)},
    },
    "ruby": {
        {re: regexp.MustCompile(`^\s*def\s`)},
        {re: regexp.MustCompile(`^\s*class\s`)},
        {re: regexp.MustCompile(`^\s*module\s`)},
    },
    "rust": {
        {re: regexp.MustCompile(`^(pub(\([^)]*\))?\s+)?(async\s+)?fn\s`)},
        {re: regexp.MustCompile(`^(pub(\([^)]*\))?\s+)?(struct|enum|trait|impl)\s`)},
    },
}

// aliases maps alternative Chroma language names to canonical keys.
var aliases = map[string]string{
    "ts":         "typescript",
    "js":         "javascript",
    "jsx":        "javascript",
    "tsx":        "typescript",
    "cpp":        "c++",
    "cxx":        "c++",
    "cc":         "c++",
    "c#":         "csharp",
    "py":         "python",
    "rb":         "ruby",
    "rs":         "rust",
}

// Find scans backward from hunkOldStart (1-indexed) in oldLines to find the
// nearest line matching a known function/class context pattern for lang.
//
// Returns the trimmed matching line content, or "" when:
//   - lang has no registered patterns (unknown language)
//   - no match is found within maxLookback lines above hunkOldStart
//   - oldLines is empty or hunkOldStart < 1
func Find(oldLines []string, hunkOldStart int, lang string) string {
    if len(oldLines) == 0 || hunkOldStart < 1 {
        return ""
    }

    key := strings.ToLower(strings.TrimSpace(lang))
    if canonical, ok := aliases[key]; ok {
        key = canonical
    }
    patterns, ok := langPatterns[key]
    if !ok {
        return ""
    }

    // Start scanning from the line just before the hunk start.
    // hunkOldStart is 1-indexed; array index is hunkOldStart-1.
    // We want the line at index hunkOldStart-2 (the line before the hunk).
    startIdx := hunkOldStart - 2
    if startIdx >= len(oldLines) {
        startIdx = len(oldLines) - 1
    }

    for i := startIdx; i >= 0 && startIdx-i < maxLookback; i-- {
        line := oldLines[i]
        trimmed := strings.TrimSpace(line)
        if trimmed == "" {
            continue
        }
        for _, p := range patterns {
            if p.re.MatchString(line) || p.re.MatchString(trimmed) {
                return strings.TrimRight(trimmed, " \t{")
            }
        }
    }
    return ""
}
```

    old := append(
        makeLines("func Alpha() {", "  x := 1", "  y := 2", "  z := 3", "  a := 4", "  b := 5", "  c := 6"),
        makeLines("  d := 7", "  e := 8", "  old := 9", "}")...,

        t.Fatalf("expected 1 hunk, got %d", len(hunks))

    if hunks[0].FuncName == "" {
        t.Error("expected FuncName to be populated for Go source")

    if !strings.HasPrefix(hunks[0].FuncName, "func Alpha") {
        t.Errorf("expected FuncName to start with 'func Alpha', got %q", hunks[0].FuncName)
