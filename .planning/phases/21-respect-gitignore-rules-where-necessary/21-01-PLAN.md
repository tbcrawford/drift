<objective>
Add `.gitignore` awareness to both directory walkers so that gitignored files are
silently skipped during directory diffs. Implements a shared `filterGitIgnored` helper
(using `git check-ignore --stdin -z`) and wires it into `gitDirectoryVsHEAD` and
`diffDirectories`.

Purpose: When users run `drift dir1/ dir2/` or `drift .` inside a git repo, build
artifacts, vendor directories, and other gitignored files should not appear in the diff
output — matching the behavior of `git diff`.

Output:
- `filterGitIgnored(repoRoot string, relPaths []string) ([]string, error)` in `cmd/drift/gitworktree.go`
- Updated `gitDirectoryVsHEAD` — filters workFiles before comparing against HEAD
- Updated `diffDirectories` — filters each side's file list independently if inside a git repo
- Tests for all filtering behaviors using fake-git scripts (consistent with existing test patterns)
</objective>

       func filterGitIgnored(repoRoot string, relPaths []string) ([]string, error) {
           if len(relPaths) == 0 {
               return nil, nil
           }
           // Build NUL-separated stdin.
           stdin := strings.Join(relPaths, "\x00") + "\x00"
           out, _, err := runGitStdin(repoRoot, stdin, "check-ignore", "-z", "--stdin")
           // Exit code 1 means "no paths ignored" — not an error.
           // Only propagate errors other than ExitError with code 1.
           // If git not found, fail open (return all paths).
           if err != nil {
               var exitErr *exec.ExitError
               if !errors.As(err, &exitErr) {
                   return relPaths, nil // git not found or other non-exit error
               }
               if exitErr.ExitCode() != 1 {
                   return relPaths, nil // unexpected error → fail open
               }
           }
           // Parse NUL-separated ignored paths into a set.
           ignoredSet := make(map[string]struct{})
           for _, p := range strings.Split(out, "\x00") {
               if p != "" {
                   ignoredSet[p] = struct{}{}
               }
           }
           // Filter: keep paths not in ignoredSet.
           var kept []string
           for _, p := range relPaths {
               if _, ignored := ignoredSet[p]; !ignored {
                   kept = append(kept, p)
               }
           }
           return kept, nil
       }
       ```

    3. Add `runGitStdin` (variant of `runGit` that accepts stdin):

       func runGitStdin(gitDir, stdin string, args ...string) (stdout, stderr string, err error) {
           gitPath, err := exec.LookPath("git")
           if err != nil {
               return "", "", err
           }
           cmd := exec.Command(gitPath, append([]string{"-C", gitDir}, args...)...)
           cmd.Env = gitEnv()
           cmd.Stdin = strings.NewReader(stdin)
           var outb, errb strings.Builder
           cmd.Stdout = &outb
           cmd.Stderr = &errb
           err = cmd.Run()
           return outb.String(), errb.String(), err
       }
       ```
