<objective>
Replace `wt.Status()` (go-git's full filesystem scan — 4,500ms) with a fast index+mtime
based change detection function that takes ~20ms.

**Root cause (measured):**
```
wt.Status():        4,579ms  ← 99% of wall-clock time
PlainOpen:              0.3ms
HEAD tree + index:      0.5ms
Index stat scan:        0.4ms
ReadFile (25 files):    1.2ms
HEAD blob read:         3.4ms
Total new approach:    23ms   ← 200x faster
```

**Why wt.Status() is slow:** go-git's `Worktree.Status()` does a full two-pass scan:
1. Walks every file in the working tree via os.ReadDir (recursive)
2. For each file, reads its SHA1 to compare against the index

On a 158-file repo, this takes 4.5 seconds. Git's own C implementation uses inode+mtime
caching (the "stat cache" in the index) to skip the SHA1 computation for files whose
mtime hasn't changed — go-git does not implement this optimization.

**New approach:** Read the git index directly (`repo.Storer.Index()`), which gives us
each file's stored mtime (`entry.ModifiedAt`). Compare against the OS mtime
(`os.Stat().ModTime()`). If they match, the file is unchanged. If they differ, include
it as a candidate. Also compare each index entry's hash against the HEAD tree hash —
staged changes are caught without any filesystem stat.

Purpose: Eliminate the dominant bottleneck so `drift --split` completes sub-500ms.
Output: `changedFilesViaIndex` function replacing the `wt.Status()` call in `gitDirectoryVsHEAD`.
</objective>

```go
func changedFilesViaIndex(repoRoot string) ([]string, error) {
    repo, err := git.PlainOpenWithOptions(repoRoot, &git.PlainOpenOptions{DetectDotGit: true})
    if err != nil {
        return nil, err
    }

    wt, err := repo.Worktree()
    if err != nil {
        return nil, err
    }
    root := wt.Filesystem.Root()

    // Get HEAD tree — empty repo (no HEAD) returns all indexed files as "new".
    var headHashes map[string]string
    ref, headErr := repo.Head()
    if headErr == nil {
        commit, cErr := repo.CommitObject(ref.Hash())
        if cErr == nil {
            headTree, tErr := commit.Tree()
            if tErr == nil {
                headHashes = make(map[string]string)
                _ = headTree.Files().ForEach(func(f *object.File) error {
                    headHashes[f.Name] = f.Hash.String()
                    return nil
                })
            }
        }
    }

    idx, err := repo.Storer.Index()
    if err != nil {
        return nil, err
    }

    var changed []string
    for _, entry := range idx.Entries {
        headHash, inHead := headHashes[entry.Name]
        if headHashes == nil || !inHead {
            // Not in HEAD → new staged file
            changed = append(changed, entry.Name)
            continue
        }
        if headHash != entry.Hash.String() {
            // Staged change (index hash differs from HEAD)
            changed = append(changed, entry.Name)
            continue
        }
        // Same as HEAD in staging — check working tree via mtime
        absPath := filepath.Join(root, filepath.FromSlash(entry.Name))
        info, err := os.Stat(absPath)
        if err != nil {
            // File deleted from working tree
            changed = append(changed, entry.Name)
            continue
        }
        if !info.ModTime().Equal(entry.ModifiedAt) {
            // mtime differs → potential unstaged change; include it
            changed = append(changed, entry.Name)
        }
    }
    return changed, nil
}
```

func BenchmarkChangedFilesViaIndex(b *testing.B) {
    // Create a 200-file repo with 20 modified files.
    files := make(map[string]string, 200)
    for i := 0; i < 200; i++ {
        files[fmt.Sprintf("dir/file%03d.go", i)] = fmt.Sprintf("package main\n// file %d\n", i)
    }
    repoDir := makeTestRepo(b.Helper(), files)  // Note: makeTestRepo needs *testing.TB

    // Modify 20 files on disk.
    for i := 0; i < 20; i++ {
        p := filepath.Join(repoDir, fmt.Sprintf("dir/file%03d.go", i))
        _ = os.WriteFile(p, []byte(fmt.Sprintf("package main\n// modified %d\n", i)), 0o644)
        // Set mtime 1s ahead to trigger change detection.
        future := time.Now().Add(time.Second)
        _ = os.Chtimes(p, future, future)
    }

    b.ResetTimer()
    for range b.N {
        changed, err := changedFilesViaIndex(repoDir)
        if err != nil {
            b.Fatal(err)
        }
        if len(changed) < 20 {
            b.Fatalf("expected ≥20 changed files, got %d", len(changed))
        }
    }
}
